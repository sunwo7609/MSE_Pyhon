apart = [ [101, 102], [201, 202], [301, 302] ]   #리스트안에 리스트가 있기 때문에 for문을 2번을 이용한다.
for 층 in apart:  #리스트apart에서 층을 for문을 이용하여 [101, 102] [201, 202] [301, 302]로 만든다.
    for 호 in 층:  #리스트층을 for문을 이용하여 호를 분리한다
        print(호,"호") # 호에 바인딩하는 값의 뒤에 문자호를 붙힌 결과값을 출력한다. 
print("-----")      #마지막으로 -----을 출력한다.
